```typescript
export async function sbGetComments(opts?: { articleId?: string; isAdmin?: boolean }): Promise<Comment[]> {
  const sb = getServiceSupabase(); // 또는 기존 패턴에 맞는 클라이언트
  let query = sb.from("comments").select("*");
  
  if (opts?.articleId) {
    query = query.eq("article_id", opts.articleId);
  }
  if (!opts?.isAdmin) {
    query = query.eq("status", "approved");
  }
  
  query = query.order("created_at", { ascending: true });
  
  const { data, error } = await query;
  if (error) throw error;
  return (data || []).map(rowToComment);
}
```

2. **sbCreateComment** — 댓글 생성
```typescript
export async function sbCreateComment(data: {
  articleId: string;
  articleTitle?: string;
  author: string;
  content: string;
  status?: Comment["status"];
  ip?: string;
  parentId?: string;
}): Promise<Comment> {
  const sb = getServiceSupabase();
  const { data: row, error } = await sb
    .from("comments")
    .insert({
      article_id: data.articleId,
      article_title: data.articleTitle || "",
      author: data.author,
      content: data.content,
      status: data.status || "approved",
      ip: data.ip || null,
      parent_id: data.parentId || null,
    })
    .select()
    .single();
  if (error) throw error;
  return rowToComment(row);
}
```

3. **sbUpdateCommentStatus** — 댓글 상태 변경

```typescript
export async function sbUpdateCommentStatus(id: string, status: Comment["status"]): Promise<void> {
  const sb = getServiceSupabase();
  const { error } = await sb
    .from("comments")
    .update({ status })
    .eq("id", id);
  if (error) throw error;
}
```

4. **sbDeleteComment** — 댓글 삭제 (자식 연쇄 삭제 포함)
```typescript
export async function sbDeleteComment(id: string): Promise<void> {
  const sb = getServiceSupabase();
  // 자식 답글 먼저 삭제 (cascade)
  const { error: childError } = await sb
    .from("comments")
    .delete()
    .eq("parent_id", id);
  if (childError) throw childError;
  // 부모 댓글 삭제
  const { error } = await sb
    .from("comments")
    .delete()
    .eq("id", id);
  if (error) throw error;
}
```
