```typescript
export async function invalidateToken(token: string): Promise<void> {
  if (redis) {
    try {
      // 토큰의 남은 유효시간(최대 24시간)동안 블랙리스트 유지
      await redis.set(`cp:token:blacklist:${token}`, "1", { ex: 86400 });
      return;
    } catch (e) {
      console.error("[cookie-auth] Redis 블랙리스트 저장 실패:", e);
    }
  }
  // Redis 없으면 인메모리 폴백 (개발환경용)
  tokenBlacklist.add(token);
}
```

3. `isTokenBlacklisted` 함수를 async로 변경하고 Redis에서 조회:
```typescript
export async function isTokenBlacklisted(token: string): Promise<boolean> {
  if (redis) {
    try {
      const result = await redis.get(`cp:token:blacklist:${token}`);
      return result !== null;
    } catch (e) {
      console.error("[cookie-auth] Redis 블랙리스트 조회 실패:", e);
      return false; // Redis 장애 시 안전하게 통과 (서비스 가용성 우선)
    }
  }
  return tokenBlacklist.has(token);
}
```

```typescript
import { NextRequest, NextResponse } from "next/server";
import { verifyAuthToken, isTokenBlacklisted } from "@/lib/cookie-auth";

export async function GET(req: NextRequest) {
  const cookie = req.cookies.get("cp-admin-auth");
  const tokenValue = cookie?.value ?? "";

  // 블랙리스트 확인 (로그아웃된 토큰 거부)
  if (tokenValue && await isTokenBlacklisted(tokenValue)) {
    return NextResponse.json({ authed: false });
  }

  const { valid, name, role } = await verifyAuthToken(tokenValue);
  if (!valid) {
    return NextResponse.json({ authed: false });
  }
  return NextResponse.json({ authed: true, name, role: role || "admin" });
}
```
